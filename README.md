# report-card-printer
# A beginner Python script demonstrating basic data types (str, bool, int, float) and how to inspect them using type() and isinstance().
name = 'Alice'
print(name, type(name))

is_student = True
print(is_student, type(is_student))

age = 20
print(age, type(age))

score = 80.5
print(isinstance(score, float))
